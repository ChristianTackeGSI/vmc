93 void TMCManager::Register(TVirtualMC *mc)
94 {
95  // Do not register an engine twice.
96  for (auto &currMC : fEngines) {
97  if (currMC == mc) {
98  ::Fatal("TMCManager::RegisterMC", "This engine is already registered.");
99  }
100  }
101  // Set id and register.
102  mc->SetId(fEngines.size());
103  fEngines.push_back(mc);
104  fStacks.emplace_back(new TMCManagerStack());
105  mc->SetStack(fStacks.back().get());
106  mc->SetManagerStack(fStacks.back().get());
107  // Must update engine pointers here since during construction of the concrete TVirtualMC
108  // implementation the static TVirtualMC::GetMC() or defined gMC might be used.
109  UpdateEnginePointers(mc);
110 }

118 void TMCManager::Register(TVirtualMCApplication *application)
119 {
120  if (fApplication) {
121  ::Fatal("TMCManager::Register", "The application is already registered.");
122  }
123  ::Info("TMCManager::Register", "Register user application and construct geometry");
124  fApplication = application;
125  // TODO Can these 3 functions can be called directly here? Or could any of these depend on an implemented VMC?
126  fApplication->ConstructGeometry();
127  fApplication->MisalignGeometry();
128  fApplication->ConstructOpGeometry();
129  if (!gGeoManager->IsClosed()) {
130  // Setting the top volume is the duty of the user as well as closing it.
131  // Failing here is just an additional cross check. If not closed the user
132  // might have forgotten something.
133  ::Fatal("TMCManager::Register", "The TGeo geometry is not closed. Please check whether you just have to close "
134  "it or whether something was forgotten.");
135  }

241 void TMCManager::ForwardTrack(Int_t toBeDone, Int_t trackId, Int_t parentId, TParticle *particle, Int_t engineId)
242 {
243  if (engineId < 0 || engineId >= static_cast<Int_t>(fEngines.size())) {
244  ::Fatal("TMCManager::ForwardTrack", "Engine ID %i out of bounds. Have %zu engines.", engineId, fEngines.size());
245  }
246  if (trackId >= static_cast<Int_t>(fParticles.size())) {
247  fParticles.resize(trackId + 1, nullptr);
248  fParticlesStatus.resize(trackId + 1);
249  }
250  fParticles[trackId] = particle;
251  fParticlesStatus[trackId].reset(new TMCParticleStatus());
252  fParticlesStatus[trackId]->fId = trackId;
253  fParticlesStatus[trackId]->fParentId = parentId;
254  fParticlesStatus[trackId]->InitFromParticle(particle);
255  fTotalNTracks++;
256  if (particle->IsPrimary()) {
257  fTotalNPrimaries++;
258  }
259 
260  if (toBeDone > 0) {
261  if (particle->IsPrimary()) {
262  fStacks[engineId]->PushPrimaryTrackId(trackId);
263  } else {
264  fStacks[engineId]->PushSecondaryTrackId(trackId);
265  }
266  }
267 }

302 void TMCManager::TransferTrack(TVirtualMC *mc)
303 {
304  // Do nothing if target and current engines are the same
305  if (mc == fCurrentEngine) {
306  return;
307  }
308 
309  // Get information on current track and extract status from transporting engine
310  Int_t trackId = fStacks[fCurrentEngine->GetId()]->GetCurrentTrackNumber();
311 
312  fCurrentEngine->TrackPosition(fParticlesStatus[trackId]->fPosition);
313  fCurrentEngine->TrackMomentum(fParticlesStatus[trackId]->fMomentum);
314  fCurrentEngine->TrackPolarization(fParticlesStatus[trackId]->fPolarization);
315  fParticlesStatus[trackId]->fStepNumber = fCurrentEngine->StepNumber();
316  fParticlesStatus[trackId]->fTrackLength = fCurrentEngine->TrackLength();
317  fParticlesStatus[trackId]->fWeight = fCurrentEngine->TrackWeight();
318 
319  // Store TGeoNavidator's fIsOutside state
320  fParticlesStatus[trackId]->fIsOutside = gGeoManager->IsOutside();
321 
322  TGeoBranchArray *geoState = fBranchArrayContainer.GetNewGeoState(fParticlesStatus[trackId]->fGeoStateIndex);
323  geoState->InitFromNavigator(gGeoManager->GetCurrentNavigator());
324 
325  // Push only the particle ID
326  if (fParticles[trackId]->IsPrimary()) {
327  fStacks[mc->GetId()]->PushPrimaryTrackId(trackId);
328  } else {
329  fStacks[mc->GetId()]->PushSecondaryTrackId(trackId);
330  }
331  fCurrentEngine->InterruptTrack();
332 }

339 Bool_t TMCManager::RestoreGeometryState(Int_t trackId, Bool_t checkTrackIdRange)
340 {
341  if (checkTrackIdRange && (trackId < 0 || trackId >= static_cast<Int_t>(fParticles.size()) || !fParticles[trackId])) {
342  return kFALSE;
343  }
344  UInt_t& geoStateId = fParticlesStatus[trackId]->fGeoStateIndex;
345  if(geoStateId == 0) {
346  return kFALSE;
347  }
348  const TGeoBranchArray* branchArray = fBranchArrayContainer.GetGeoState(geoStateId);
349  branchArray->UpdateNavigator(gGeoManager->GetCurrentNavigator());
350  fBranchArrayContainer.FreeGeoState(geoStateId);
351  gGeoManager->SetOutside(fParticlesStatus[trackId]->fIsOutside);
352  geoStateId = 0;
353  return kTRUE;
354 }

371 void TMCManager::Init()
372 {
373  if (fIsInitialized) {
374  return;
375  }
376 
377  if (!fUserStack) {
378  ::Fatal("TMCManager::Run", "Missing user stack pointer.");
379  }
380  if (fEngines.empty()) {
381  ::Fatal("TMCManager::Run", "No engines registered");
382  }
383 
384  for (auto &mc : fEngines) {
385  // Must have geometry handling via TGeo
386  if (!mc->IsRootGeometrySupported()) {
387  ::Fatal("TMCManager::Run", "Engine %s does not support geometry built via ROOT's TGeoManager", mc->GetName());
388  }
389  Int_t currentEngineId = mc->GetId();
390  // To be able to forward info to the user stack
391  fStacks[currentEngineId]->SetUserStack(fUserStack);
392  // Connect the engine's stack to the centrally managed vectors
393  fStacks[currentEngineId]->ConnectTrackContainers(&fParticles, &fParticlesStatus, &fBranchArrayContainer,
394  &fTotalNPrimaries, &fTotalNTracks);
395  }
396 
397  // Initialize the fBranchArrayContainer to manage and cache TGeoBranchArrays
398  fBranchArrayContainer.InitializeFromGeoManager(gGeoManager);
399 
400  fIsInitialized = kTRUE;
401 
402  // Send warning if only one engine ==> overhead
403  if (fEngines.size() == 1) {
404  ::Warning("TMCManager::Run", "Only one engine registered. That will lead to overhead in "
405  "the simulation run due to additional hooks and dispatches "
406  "to the TMCManager.");
407  }
408 }

415 void TMCManager::Run(Int_t nEvents)
416 {
417  if (!fIsInitialized) {
418  ::Fatal("TMCManager::Run", "Engines have not yet been initialized.");
419  }
420 
421  // Set user initialize true in any case now so that this cannot happen by accident during a run
422  fIsInitializedUser = kTRUE;
423 
424  if (nEvents < 1) {
425  ::Fatal("TMCManager::Run", "Need at least one event to process but %i events specified.", nEvents);
426  }
427 
428  // Run 1 event nEvents times
429  for (Int_t i = 0; i < nEvents; i++) {
430  ::Info("TMCManager::Run", "Start event %i", i + 1);
431  PrepareNewEvent();
432  fApplication->BeginEvent();
433  // Loop as long as there are tracks in any engine stack
434  while (GetNextEngine()) {
435  fCurrentEngine->ProcessEvent(i, kTRUE);
436  }
437  fApplication->FinishEvent();
438  }
439  TerminateRun();
440 }

447 void TMCManager::PrepareNewEvent()
448 {
449  fBranchArrayContainer.FreeGeoStates();
450  // Reset in event flag for all engines and clear stacks
451  for (auto &stack : fStacks) {
452  stack->ResetInternals();
453  }
454  for (UInt_t i = 0; i < fParticles.size(); i++) {
455  fParticlesStatus.clear();
456  fParticlesStatus.resize(fParticles.size());
457  fParticles[i] = nullptr;
458  }
459 
460  // GeneratePrimaries centrally
461  fApplication->GeneratePrimaries();
462 }

470 {

472  for (UInt_t i = 0; i < fStacks.size(); i++) {
473  if (fStacks[i]->GetStackedNtrack() > 0) {
