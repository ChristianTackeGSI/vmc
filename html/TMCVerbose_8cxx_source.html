249 void TMCVerbose::Stepping()
250 {
251  if (fLevel>2) {
252 
253 #if __GNUC__ >= 3
254  std::cout << std::fixed;
255 #endif
256 
257  // Step number
258  //
259  // std::cout << "#" << std::setw(4) << gMC->StepNumber() << " ";
260  std::cout << "#" << std::setw(4) << fStepNumber++ << " ";
261 
262  // Position
263  //
264  Double_t x, y, z;
265  gMC->TrackPosition(x, y, z);
266  std::cout << std::setw(8) << std::setprecision(3) << x << " "
267  << std::setw(8) << std::setprecision(3) << y << " "
268  << std::setw(8) << std::setprecision(3) << z << " ";
269 
270  // Kinetic energy
271  //
272  Double_t px, py, pz, etot;
273  gMC->TrackMomentum(px, py, pz, etot);
274  Double_t ekin = etot - gMC->TrackMass();
275  std::cout << std::setw(9) << std::setprecision(4) << ekin*1e03 << " ";
276 
277  // Energy deposit
278  //
279  std::cout << std::setw(9) << std::setprecision(4) << gMC->Edep()*1e03 << " ";
280 
281  // Step length
282  //
283  std::cout << std::setw(8) << std::setprecision(3) << gMC->TrackStep() << " ";
284 
285  // Track length
286  //
287  std::cout << std::setw(8) << std::setprecision(3) << gMC->TrackLength() << " ";
288 
289  // Volume
290  //
291  if (gMC->CurrentVolName() != 0)
292  std::cout << std::setw(4) << gMC->CurrentVolName() << " ";
293  else
294  std::cout << std::setw(4) << "None" << " ";
295 
296  // Process
297  //
298  TArrayI processes;
299  Int_t nofProcesses = gMC->StepProcesses(processes);
300  if (nofProcesses > 0)
301  std::cout << TMCProcessName[processes[nofProcesses-1]];
302 
303  std::cout << std::endl;
304  }
305 }
