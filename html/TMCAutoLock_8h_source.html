66 // Class Description:
67 //
68 // This class provides a mechanism to create a mutex and locks/unlocks it.
69 // Can be used by applications to implement in a portable way a mutexing logic.
70 // Usage Example:
71 //
72 // #include "G4Threading.hh"
73 // #include "G4AutoLock.hh"
74 // /* somehwere */
75 // G4Mutex aMutex = G4MUTEX_INITIALIZER;
76 // /*
77 // somewhere else:
78 // The G4AutoLock instance will automatically unlock the mutex when it
79 // goes out of scope, lock and unlock method are anyway available for
80 // explicit handling of mutex lock. */
81 // G4AutoLock l(&aMutex);
82 // ProtectedCode();
83 // l.unlock(); //explicit unlock
84 // UnprotectedCode();
85 // l.lock(); //explicit lock
86 //
87 // Note that G4AutoLock is defined also for a sequential Geant4 build,
88 // but has no effect.

124 template <class M, typename L, typename U>
125 class TMCTemplateAutoLock {
126 public:
127  TMCTemplateAutoLock(M *mtx, L l, U u) : locked(false), _m(mtx), _l(l), _u(u) { lock(); }
128 
129  virtual ~TMCTemplateAutoLock() { unlock(); }
130 
131  inline void unlock()
132  {
133  if (!locked) return;
134  _u(_m);
135  locked = false;
136  }
137 
138  inline void lock()
139  {
140  if (locked) return;
141  _l(_m);
142  locked = true;
143  }
144 
145 private:
146  // Disable copy and assignement operators
147  //
148  TMCTemplateAutoLock(const TMCTemplateAutoLock &rhs);
149  TMCTemplateAutoLock &operator=(const TMCTemplateAutoLock &rhs);
150 
151 private:
152  bool locked;
153  M *_m;
154  L _l;
155  U _u;
156 };
